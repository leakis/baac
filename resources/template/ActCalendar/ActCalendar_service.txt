      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin ActCalendar

       public int GetActCalendarTotalCount(GameId gameId)
{
Expression<Func<ActCalendar, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.ActCalendar.Where(predicate).Count();
return totalCount;
}
public List<ActCalendarAdminModel> GetActCalendarList(GameId gameId,int skip, int pageSize)
{
{List<ActCalendar> list = null;
Expression<Func<ActCalendar, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.ActCalendar.Where(predicate).OrderByDescending(a => a.CalendarId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<ActCalendar>, List<ActCalendarAdminModel>>(list);
return modellist;
}


       

		public bool DeleteActCalendar(int id, int m)
        { 
               var actcalendar=new ActCalendar(){ CalendarId=id, IsDelete=m };
               var entry = context.Entry(actcalendar);
               context.Set<ActCalendar>().Attach(actcalendar);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveActCalendar(ActCalendarAdminModel data)
        {
           var domaindata=  AutoMapper.To<ActCalendarAdminModel, ActCalendar>(data);
            context.ActCalendar.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModActCalendar(ActCalendarAdminModel data)
        {
            var domaindata=  AutoMapper.To<ActCalendarAdminModel, ActCalendar>(data);
            var entry = context.Entry(domaindata);
            context.Set<ActCalendar>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public ActCalendar GetActCalendar(int id)
        {
            var model = context.ActCalendar.Find(id);
            return model;
        }



		#endregion end ActCalendar

		}
