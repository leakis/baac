		#region begin SendGiftLog

		public ActionResult SendGiftLogList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetSendGiftLogTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<SendGiftLogAdminModel> list =_service.GetSendGiftLogList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<SendGiftLog>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetSendGiftLog_Ajax(int id)
        {
            var rtn = _service.GetSendGiftLog(id);

        }

             log.InfoFormat("用户：{0}, 修改SendGiftLog: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;

            }

            {

            }
             return Json(rtn);
         }

		        //删除
