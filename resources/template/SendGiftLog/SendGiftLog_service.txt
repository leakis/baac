      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin SendGiftLog

       public int GetSendGiftLogTotalCount(GameId gameId)
{
Expression<Func<SendGiftLog, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.SendGiftLog.Where(predicate).Count();
return totalCount;
}
public List<SendGiftLogAdminModel> GetSendGiftLogList(GameId gameId,int skip, int pageSize)
{
{List<SendGiftLog> list = null;
Expression<Func<SendGiftLog, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.SendGiftLog.Where(predicate).OrderByDescending(a => a.LogId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<SendGiftLog>, List<SendGiftLogAdminModel>>(list);
return modellist;
}


       

		public bool DeleteSendGiftLog(int id, int m)
        { 
               var sendgiftlog=new SendGiftLog(){ LogId=id, IsDelete=m };
               var entry = context.Entry(sendgiftlog);
               context.Set<SendGiftLog>().Attach(sendgiftlog);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveSendGiftLog(SendGiftLogAdminModel data)
        {
           var domaindata=  AutoMapper.To<SendGiftLogAdminModel, SendGiftLog>(data);
            context.SendGiftLog.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModSendGiftLog(SendGiftLogAdminModel data)
        {
            var domaindata=  AutoMapper.To<SendGiftLogAdminModel, SendGiftLog>(data);
            var entry = context.Entry(domaindata);
            context.Set<SendGiftLog>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public SendGiftLog GetSendGiftLog(int id)
        {
            var model = context.SendGiftLog.Find(id);
            return model;
        }



		#endregion end SendGiftLog

		}
