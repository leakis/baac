      [Auth]
    public class ##Controller : ControllerBase
    {
        private static readonly log4net.ILog log = log4net.LogManager.GetLogger(typeof(##Controller));
        ##Service _service = null;
        //
        // GET: /SystemCenter/
        public ##Controller()
        {
            _service = new ##Service();
        }




		#region begin RuleEvent

		public ActionResult RuleEventList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetRuleEventTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<RuleEventAdminModel> list =_service.GetRuleEventList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<RuleEvent>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetRuleEvent_Ajax(int id)
        {
            var rtn = _service.GetRuleEvent(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveRuleEvent_Ajax(RuleEventAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改RuleEvent: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModRuleEvent(data); 
            }
            else
            {
                 rtn = _service.SaveRuleEvent(data);
            }
             return Json(rtn);
         }

		        //删除
        [HttpPost]
        public JsonResult DeleteRuleEvent_Ajax(int id, int m)
        {  log.InfoFormat("用户：{0}, 修改RuleEvent: id:{1}, m:{2} ", Session["AdminUser"], id,m);
            bool isSuc = _service.DeleteRuleEvent(id, m);
            return Json(isSuc);
        }
