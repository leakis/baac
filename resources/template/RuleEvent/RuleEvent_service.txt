      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin RuleEvent

       public int GetRuleEventTotalCount(GameId gameId)
{
Expression<Func<RuleEvent, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.RuleEvent.Where(predicate).Count();
return totalCount;
}
public List<RuleEventAdminModel> GetRuleEventList(GameId gameId,int skip, int pageSize)
{
{List<RuleEvent> list = null;
Expression<Func<RuleEvent, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.RuleEvent.Where(predicate).OrderByDescending(a => a.EventId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<RuleEvent>, List<RuleEventAdminModel>>(list);
return modellist;
}


       

		public bool DeleteRuleEvent(int id, int m)
        { 
               var ruleevent=new RuleEvent(){ EventId=id, IsDelete=m };
               var entry = context.Entry(ruleevent);
               context.Set<RuleEvent>().Attach(ruleevent);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveRuleEvent(RuleEventAdminModel data)
        {
           var domaindata=  AutoMapper.To<RuleEventAdminModel, RuleEvent>(data);
            context.RuleEvent.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModRuleEvent(RuleEventAdminModel data)
        {
            var domaindata=  AutoMapper.To<RuleEventAdminModel, RuleEvent>(data);
            var entry = context.Entry(domaindata);
            context.Set<RuleEvent>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public RuleEvent GetRuleEvent(int id)
        {
            var model = context.RuleEvent.Find(id);
            return model;
        }



		#endregion end RuleEvent

		}
