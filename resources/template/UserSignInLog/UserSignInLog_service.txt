      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin UserSignInLog

       public int GetUserSignInLogTotalCount(GameId gameId)
{
Expression<Func<UserSignInLog, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.UserSignInLog.Where(predicate).Count();
return totalCount;
}
public List<UserSignInLogAdminModel> GetUserSignInLogList(GameId gameId,int skip, int pageSize)
{
{List<UserSignInLog> list = null;
Expression<Func<UserSignInLog, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.UserSignInLog.Where(predicate).OrderByDescending(a => a.LogId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<UserSignInLog>, List<UserSignInLogAdminModel>>(list);
return modellist;
}


       

		public bool DeleteUserSignInLog(int id, int m)
        { 
               var usersigninlog=new UserSignInLog(){ LogId=id, IsDelete=m };
               var entry = context.Entry(usersigninlog);
               context.Set<UserSignInLog>().Attach(usersigninlog);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveUserSignInLog(UserSignInLogAdminModel data)
        {
           var domaindata=  AutoMapper.To<UserSignInLogAdminModel, UserSignInLog>(data);
            context.UserSignInLog.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModUserSignInLog(UserSignInLogAdminModel data)
        {
            var domaindata=  AutoMapper.To<UserSignInLogAdminModel, UserSignInLog>(data);
            var entry = context.Entry(domaindata);
            context.Set<UserSignInLog>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public UserSignInLog GetUserSignInLog(int id)
        {
            var model = context.UserSignInLog.Find(id);
            return model;
        }



		#endregion end UserSignInLog

		}
