		#region begin UserSignInLog

		public ActionResult UserSignInLogList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserSignInLogTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserSignInLogAdminModel> list =_service.GetUserSignInLogList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserSignInLog>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserSignInLog_Ajax(int id)
        {
            var rtn = _service.GetUserSignInLog(id);

            return JsonConvert.SerializeObject(rtn);
        }

         public JsonResult SaveUserSignInLog_Ajax(UserSignInLogAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改UserSignInLog: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModUserSignInLog(data); 
            }
            else
            {
                 rtn = _service.SaveUserSignInLog(data);
            }
             return Json(rtn);
         }

		        //删除
