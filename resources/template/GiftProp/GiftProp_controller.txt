      [Auth]
    public class ##Controller : ControllerBase
    {
        private static readonly log4net.ILog log = log4net.LogManager.GetLogger(typeof(##Controller));
        ##Service _service = null;
        //
        // GET: /SystemCenter/
        public ##Controller()
        {
            _service = new ##Service();
        }




		#region begin GiftProp

		public ActionResult GiftPropList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetGiftPropTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<GiftPropAdminModel> list =_service.GetGiftPropList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<GiftProp>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetGiftProp_Ajax(int id)
        {
            var rtn = _service.GetGiftProp(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveGiftProp_Ajax(GiftPropAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改GiftProp: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModGiftProp(data); 
            }
            else
            {
                 rtn = _service.SaveGiftProp(data);
            }
             return Json(rtn);
         }

		        //删除
        [HttpPost]
        public JsonResult DeleteGiftProp_Ajax(int id, int m)
        {  log.InfoFormat("用户：{0}, 修改GiftProp: id:{1}, m:{2} ", Session["AdminUser"], id,m);
            bool isSuc = _service.DeleteGiftProp(id, m);
            return Json(isSuc);
        }

		 #endregion end GiftProp


   }
