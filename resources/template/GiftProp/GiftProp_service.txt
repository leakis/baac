      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin GiftProp

       public int GetGiftPropTotalCount(GameId gameId)
{
Expression<Func<GiftProp, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.GiftProp.Where(predicate).Count();
return totalCount;
}
public List<GiftPropAdminModel> GetGiftPropList(GameId gameId,int skip, int pageSize)
{
{List<GiftProp> list = null;
Expression<Func<GiftProp, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.GiftProp.Where(predicate).OrderByDescending(a => a.GiftId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<GiftProp>, List<GiftPropAdminModel>>(list);
return modellist;
}


       

		public bool DeleteGiftProp(int id, int m)
        { 
               var giftprop=new GiftProp(){ GiftId=id, IsDelete=m };
               var entry = context.Entry(giftprop);
               context.Set<GiftProp>().Attach(giftprop);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveGiftProp(GiftPropAdminModel data)
        {
           var domaindata=  AutoMapper.To<GiftPropAdminModel, GiftProp>(data);
            context.GiftProp.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModGiftProp(GiftPropAdminModel data)
        {
            var domaindata=  AutoMapper.To<GiftPropAdminModel, GiftProp>(data);
            var entry = context.Entry(domaindata);
            context.Set<GiftProp>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public GiftProp GetGiftProp(int id)
        {
            var model = context.GiftProp.Find(id);
            return model;
        }



		#endregion end GiftProp

		}
