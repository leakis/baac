      [Auth]
    public class ##Controller : ControllerBase
    {
        private static readonly log4net.ILog log = log4net.LogManager.GetLogger(typeof(##Controller));
        ##Service _service = null;
        //
        // GET: /SystemCenter/
        public ##Controller()
        {
            _service = new ##Service();
        }




		#region begin AutoReplyRule

		public ActionResult AutoReplyRuleList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetAutoReplyRuleTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<AutoReplyRuleAdminModel> list =_service.GetAutoReplyRuleList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<AutoReplyRule>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetAutoReplyRule_Ajax(int id)
        {
            var rtn = _service.GetAutoReplyRule(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveAutoReplyRule_Ajax(AutoReplyRuleAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改AutoReplyRule: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModAutoReplyRule(data); 
            }
            else
            {
                 rtn = _service.SaveAutoReplyRule(data);
            }
             return Json(rtn);
         }

		        //删除
        [HttpPost]
        public JsonResult DeleteAutoReplyRule_Ajax(int id, int m)
        {  log.InfoFormat("用户：{0}, 修改AutoReplyRule: id:{1}, m:{2} ", Session["AdminUser"], id,m);
            bool isSuc = _service.DeleteAutoReplyRule(id, m);
            return Json(isSuc);
        }
