      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin UserSubscribeLog

       public int GetUserSubscribeLogTotalCount(GameId gameId)
{
Expression<Func<UserSubscribeLog, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.UserSubscribeLog.Where(predicate).Count();
return totalCount;
}
public List<UserSubscribeLogAdminModel> GetUserSubscribeLogList(GameId gameId,int skip, int pageSize)
{
{List<UserSubscribeLog> list = null;
Expression<Func<UserSubscribeLog, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.UserSubscribeLog.Where(predicate).OrderByDescending(a => a.LogId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<UserSubscribeLog>, List<UserSubscribeLogAdminModel>>(list);
return modellist;
}


       

		public bool DeleteUserSubscribeLog(int id, int m)
        { 
               var usersubscribelog=new UserSubscribeLog(){ LogId=id, IsDelete=m };
               var entry = context.Entry(usersubscribelog);
               context.Set<UserSubscribeLog>().Attach(usersubscribelog);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveUserSubscribeLog(UserSubscribeLogAdminModel data)
        {
           var domaindata=  AutoMapper.To<UserSubscribeLogAdminModel, UserSubscribeLog>(data);
            context.UserSubscribeLog.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModUserSubscribeLog(UserSubscribeLogAdminModel data)
        {
            var domaindata=  AutoMapper.To<UserSubscribeLogAdminModel, UserSubscribeLog>(data);
            var entry = context.Entry(domaindata);
            context.Set<UserSubscribeLog>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public UserSubscribeLog GetUserSubscribeLog(int id)
        {
            var model = context.UserSubscribeLog.Find(id);
            return model;
        }



		#endregion end UserSubscribeLog

		}
