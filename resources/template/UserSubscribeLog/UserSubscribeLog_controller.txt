        ##Service _service = null;

        // GET: /SystemCenter/
        public ##Controller()
        {
            _service = new ##Service();
        }




		#region begin UserSubscribeLog

		public ActionResult UserSubscribeLogList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserSubscribeLogTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserSubscribeLogAdminModel> list =_service.GetUserSubscribeLogList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserSubscribeLog>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserSubscribeLog_Ajax(int id)
        {
            var rtn = _service.GetUserSubscribeLog(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveUserSubscribeLog_Ajax(UserSubscribeLogAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改UserSubscribeLog: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModUserSubscribeLog(data); 
            }
            else
            {
                 rtn = _service.SaveUserSubscribeLog(data);
            }
             return Json(rtn);
         }

		        //删除
        [HttpPost]
        public JsonResult DeleteUserSubscribeLog_Ajax(int id, int m)
        {  log.InfoFormat("用户：{0}, 修改UserSubscribeLog: id:{1}, m:{2} ", Session["AdminUser"], id,m);
            bool isSuc = _service.DeleteUserSubscribeLog(id, m);
            return Json(isSuc);
        }
