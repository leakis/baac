      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin ActUserAttention

       public int GetActUserAttentionTotalCount(GameId gameId)
{
Expression<Func<ActUserAttention, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.ActUserAttention.Where(predicate).Count();
return totalCount;
}
public List<ActUserAttentionAdminModel> GetActUserAttentionList(GameId gameId,int skip, int pageSize)
{
{List<ActUserAttention> list = null;
Expression<Func<ActUserAttention, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.ActUserAttention.Where(predicate).OrderByDescending(a => a.).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<ActUserAttention>, List<ActUserAttentionAdminModel>>(list);
return modellist;
}


       

		public bool DeleteActUserAttention(int id, int m)
        { 
               var actuserattention=new ActUserAttention(){ =id, IsDelete=m };
               var entry = context.Entry(actuserattention);
               context.Set<ActUserAttention>().Attach(actuserattention);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveActUserAttention(ActUserAttentionAdminModel data)
        {
           var domaindata=  AutoMapper.To<ActUserAttentionAdminModel, ActUserAttention>(data);
            context.ActUserAttention.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModActUserAttention(ActUserAttentionAdminModel data)
        {
            var domaindata=  AutoMapper.To<ActUserAttentionAdminModel, ActUserAttention>(data);
            var entry = context.Entry(domaindata);
            context.Set<ActUserAttention>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public ActUserAttention GetActUserAttention(int id)
        {
            var model = context.ActUserAttention.Find(id);
            return model;
        }



		#endregion end ActUserAttention

		}
