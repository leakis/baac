		#region begin ActUserAttention

		public ActionResult ActUserAttentionList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetActUserAttentionTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<ActUserAttentionAdminModel> list =_service.GetActUserAttentionList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<ActUserAttention>();
            }
            return View(list);
        }
