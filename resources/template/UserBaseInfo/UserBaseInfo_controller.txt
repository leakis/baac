		#region begin UserBaseInfo

		public ActionResult UserBaseInfoList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserBaseInfoTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserBaseInfoAdminModel> list =_service.GetUserBaseInfoList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserBaseInfo>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserBaseInfo_Ajax(int id)
        {
            var rtn = _service.GetUserBaseInfo(id);

            return JsonConvert.SerializeObject(rtn);
        }

         {
             log.InfoFormat("用户：{0}, 修改UserBaseInfo: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;

            {
                 rtn = _service.ModUserBaseInfo(data);

            else
            {
                 rtn = _service.SaveUserBaseInfo(data);
            }
             return Json(rtn);
         }
