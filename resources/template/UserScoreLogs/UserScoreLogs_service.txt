      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin UserScoreLogs

       public int GetUserScoreLogsTotalCount(GameId gameId)
{
Expression<Func<UserScoreLogs, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.UserScoreLogs.Where(predicate).Count();
return totalCount;
}
public List<UserScoreLogsAdminModel> GetUserScoreLogsList(GameId gameId,int skip, int pageSize)
{
{List<UserScoreLogs> list = null;
Expression<Func<UserScoreLogs, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.UserScoreLogs.Where(predicate).OrderByDescending(a => a.LogId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<UserScoreLogs>, List<UserScoreLogsAdminModel>>(list);
return modellist;
}


       

		public bool DeleteUserScoreLogs(int id, int m)
        { 
               var userscorelogs=new UserScoreLogs(){ LogId=id, IsDelete=m };
               var entry = context.Entry(userscorelogs);
               context.Set<UserScoreLogs>().Attach(userscorelogs);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveUserScoreLogs(UserScoreLogsAdminModel data)
        {
           var domaindata=  AutoMapper.To<UserScoreLogsAdminModel, UserScoreLogs>(data);
            context.UserScoreLogs.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModUserScoreLogs(UserScoreLogsAdminModel data)
        {
            var domaindata=  AutoMapper.To<UserScoreLogsAdminModel, UserScoreLogs>(data);
            var entry = context.Entry(domaindata);
            context.Set<UserScoreLogs>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public UserScoreLogs GetUserScoreLogs(int id)
        {
            var model = context.UserScoreLogs.Find(id);
            return model;
        }



		#endregion end UserScoreLogs

		}
