      [Auth]
    public class ##Controller : ControllerBase
    {
        private static readonly log4net.ILog log = log4net.LogManager.GetLogger(typeof(##Controller));
        ##Service _service = null;
        //
        // GET: /SystemCenter/
        public ##Controller()
        {
            _service = new ##Service();
        }




		#region begin UserScoreLogs

		public ActionResult UserScoreLogsList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserScoreLogsTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserScoreLogsAdminModel> list =_service.GetUserScoreLogsList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserScoreLogs>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserScoreLogs_Ajax(int id)
        {
            var rtn = _service.GetUserScoreLogs(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveUserScoreLogs_Ajax(UserScoreLogsAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改UserScoreLogs: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModUserScoreLogs(data); 
            }
            else
            {
                 rtn = _service.SaveUserScoreLogs(data);
            }
             return Json(rtn);
         }

		        //删除
        [HttpPost]
        public JsonResult DeleteUserScoreLogs_Ajax(int id, int m)
        {  log.InfoFormat("用户：{0}, 修改UserScoreLogs: id:{1}, m:{2} ", Session["AdminUser"], id,m);
            bool isSuc = _service.DeleteUserScoreLogs(id, m);
            return Json(isSuc);
        }
