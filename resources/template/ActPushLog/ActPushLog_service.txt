      
 public class SystemCenterService
    {
	   WeiXinEntities context = new WeiXinEntities();

	   #region begin ActPushLog

       public int GetActPushLogTotalCount(GameId gameId)
{
Expression<Func<ActPushLog, bool>> predicate = a => a.GameId == (long)gameId;
int totalCount = 0;
totalCount = context.ActPushLog.Where(predicate).Count();
return totalCount;
}
public List<ActPushLogAdminModel> GetActPushLogList(GameId gameId,int skip, int pageSize)
{
{List<ActPushLog> list = null;
Expression<Func<ActPushLog, bool>> predicate = a => a.GameId == (long)gameId;
list =  context.ActPushLog.Where(predicate).OrderByDescending(a => a.LogId).Skip(skip).Take(pageSize).ToList();
var modellist = AutoMapper.To<List<ActPushLog>, List<ActPushLogAdminModel>>(list);
return modellist;
}


       

		public bool DeleteActPushLog(int id, int m)
        { 
               var actpushlog=new ActPushLog(){ LogId=id, IsDelete=m };
               var entry = context.Entry(actpushlog);
               context.Set<ActPushLog>().Attach(actpushlog);
               entry.Property("IsDelete").IsModified = true;
              int rtn= context.SaveChanges();
              return rtn > 0;
        }


		 public bool SaveActPushLog(ActPushLogAdminModel data)
        {
           var domaindata=  AutoMapper.To<ActPushLogAdminModel, ActPushLog>(data);
            context.ActPushLog.Add(domaindata);
            var rtn= context.SaveChanges();
            return true;
        }

		 public bool ModActPushLog(ActPushLogAdminModel data)
        {
            var domaindata=  AutoMapper.To<ActPushLogAdminModel, ActPushLog>(data);
            var entry = context.Entry(domaindata);
            context.Set<ActPushLog>().Attach(domaindata);
            entry.State = EntityState.Modified;
            int rtn = context.SaveChanges();
            return rtn > 0;
        }

	    public ActPushLog GetActPushLog(int id)
        {
            var model = context.ActPushLog.Find(id);
            return model;
        }



		#endregion end ActPushLog

		}
