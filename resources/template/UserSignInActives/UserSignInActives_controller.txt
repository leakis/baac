        public ##Controller()

		#region begin UserSignInActives

		public ActionResult UserSignInActivesList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserSignInActivesTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserSignInActivesAdminModel> list =_service.GetUserSignInActivesList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserSignInActives>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserSignInActives_Ajax(int id)
        {
            var rtn = _service.GetUserSignInActives(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveUserSignInActives_Ajax(UserSignInActivesAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改UserSignInActives: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModUserSignInActives(data); 
            }
            else
            {
                 rtn = _service.SaveUserSignInActives(data);
            }
             return Json(rtn);
         }

		        //删除
