		#region begin UserGameCharacter

		public ActionResult UserGameCharacterList(int page=1)
        {
            int p = page <= 0 ? 1 : page;
            int pageSize = 100;
            int skip = (p - 1) * pageSize;
            int totalCount =_service.GetUserGameCharacterTotalCount(GameId);
            Pagination pagination = new Pagination(totalCount, p, pageSize);
            ViewBag.Pagination = pagination;
            List<UserGameCharacterAdminModel> list =_service.GetUserGameCharacterList(GameId,skip,pageSize);
            if (list == null)
            {
                list = new List<UserGameCharacter>();
            }
            return View(list);
        }

         [HttpGet]
        public string GetUserGameCharacter_Ajax(int id)
        {
            var rtn = _service.GetUserGameCharacter(id);

            return JsonConvert.SerializeObject(rtn);
        }

         [HttpPost]
         public JsonResult SaveUserGameCharacter_Ajax(UserGameCharacterAdminModel data)
         {
             log.InfoFormat("用户：{0}, 修改UserGameCharacter: {1} ", Session["AdminUser"], JsonConvert.SerializeObject(data));
			 bool rtn = false;
            if (data.{{pk}} > 0)
            {
                 rtn = _service.ModUserGameCharacter(data); 
            }
            else
            {
                 rtn = _service.SaveUserGameCharacter(data);
            }
             return Json(rtn);
         }
